  'USEUNIT ReportPortalApiUtils 

'USEUNIT globalVariables

Class clsReportPortalApiHandler

Dim objReportPortalApiUtils

Private baseUrl, authToken

Public Sub Init(url, token)

baseUrl = url

authToken = token

Set objReportPortalApiUtils = initialiseclsReportPortalApiUtils()

End Sub

' Make escaping a bit more readable :-S

Public Function Qu(ByVal s)

Qu = Null

If (VarType(s) = vbString) Then

Qu = Chr(34) & CStr(s) & Chr(34)

End If

End Function

Public Function MakeRequest(url, requestType, data)

Set objHTTP = CreateObject("MSXML2.ServerXMLHTTP")

objHTTP.SetTimeouts 0, 60000, 300000, 300000

objHTTP.Open requestType, url, False

objHTTP.setRequestHeader "User-Agent", "Mozilla/4.0"

'authToken = "bearer 4a15a7cf-975d-422d-8f27-abafbff1ef44" 'TODO: Remove

objHTTP.setRequestHeader "Authorization", authToken

objHTTP.setRequestHeader "Content-Type", "application/json;"

objHTTP.setRequestHeader "CharSet", "charset=UTF-8"

objHTTP.setRequestHeader "Accept", "application/json"

objHTTP.send (data)

Set MakeRequest = objHTTP

End Function

Public Function MakeMultipartRequest(url, requestType, data, strBoundary)

Set objHTTP = CreateObject("MSXML2.ServerXMLHTTP")

objHTTP.SetTimeouts 0, 60000, 300000, 300000

objHTTP.Open requestType, url, False

objHTTP.setRequestHeader "User-Agent", "Mozilla/4.0"

'authToken = "bearer 4a15a7cf-975d-422d-8f27-abafbff1ef44" 'TODO: Remove

objHTTP.setRequestHeader "Authorization", authToken

objHTTP.SetRequestHeader "Content-type", "multipart/form-data; boundary=" & strBoundary

objHTTP.SetRequestHeader "Content-Length", Len(data)

objHTTP.setRequestHeader "CharSet", "charset=UTF-8"

objHTTP.setRequestHeader "Accept", "application/json"

objHTTP.send (data)

Set MakeMultipartRequest = objHTTP

End Function

Public Function RpLaunchStart(ByRef returnValue, launchName, launchDesc)

If(rpdebug = true) Then

RpLaunchStart = False

Exit Function

End if

RpLaunchStart = False

call objReportPortalApiUtils.WriteConfig("", False)

jsonData = " { " &_

Qu("description") & ": " & Qu(launchDesc) & " , " &_

Qu("mode") & ": " & Qu(launchTab) & " , " &_

Qu("name") & ": " & Qu(launchName) & ", " &_

Qu("start_time") & ": " & Qu(objReportPortalApiUtils.GetTimeStamp()) & ", " &_

Qu("tags") & ": [ " & Qu("CAMA1") & ", " & Qu("CAMA2") & " ] " &_

" } "

url = baseUrl

'url = "http://35.163.39.80:8080/api/v1/aumentum"

url = url & "/launch"

Set objHTTP = MakeRequest(url, "POST", jsonData)

If objHTTP.Status >= 400 And objHTTP.Status <= 599 Then

returnValue = "Error|" & CInt(objHTTP.Status) & "|" & objHTTP.statusText & "|" & objHTTP.ResponseText

Else

returnValue = "Success|" & CInt(objHTTP.Status) & "|" & objHTTP.ResponseText

writeValue = objReportPortalApiUtils.GetIdPair(objHTTP.ResponseText, "launchid")

call objReportPortalApiUtils.WriteConfig(writeValue, False)

RpLaunchStart = True

End If

'call objReportPortalApiUtils.WriteConfig("launchid:gfdgfHGFFHFGFHJGF", False) 'TOBe removed

End Function
