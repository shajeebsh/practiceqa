    -- Best for right-skewed insurance data

        SELECT
            claim_type,
            PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY claim_amount) AS q1,
            PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY claim_amount) AS q3

        WHERE claim_amount IS NOT NULL AND claim_amount > 0
        GROUP BY claim_type
    ),
    bounds AS (
        SELECT claim_type, q1, q3,
               ROUND(q1 - 1.5*(q3-q1), 2) AS lower_fence,
               ROUND(q3 + 1.5*(q3-q1), 2) AS upper_fence
        FROM quartiles
    )

           b.lower_fence, b.upper_fence, 'OUTLIER' AS flag
    FROM claims c
    JOIN bounds b ON c.claim_type = b.claim_type
    WHERE (c.claim_amount > b.upper_fence OR c.claim_amount < b.lower_fence)
      AND c.claim_amount IS NOT NULL
    ORDER BY c.claim_amount DESC;

    -- ── 2B. Z-SCORE METHOD ────────────────────────────────────────
    -- Better for normally distributed data; less robust for skewed
     
    WITH stats AS (
        SELECT AVG(claim_amount) AS mean_amt,
               STDEV(claim_amount) AS std_amt
               -- SQLite: SQRT(AVG(claim_amount*claim_amount) - AVG(claim_amount)*AVG(claim_amount))
        FROM claims
        WHERE claim_amount IS NOT NULL AND claim_amount > 0
    )
    SELECT c.claim_id, c.claim_type,
           ROUND(c.claim_amount, 2) AS claim_amount,
           ROUND(ABS(c.claim_amount - s.mean_amt) / NULLIF(s.std_amt, 0), 2) AS z_score
    FROM claims c CROSS JOIN stats s
    WHERE c.claim_amount IS NOT NULL AND c.claim_amount > 0
      AND ABS(c.claim_amount - s.mean_amt) / NULLIF(s.std_amt, 0) > 3
    ORDER BY z_score DESC;

    -- Preferred approach in insurance — keep data, add signal
     
    WITH bounds AS (
        SELECT
            PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY claim_amount)
                - 1.5 * (PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY claim_amount)
                       - PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY claim_amount)) AS lower_fence,
            PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY claim_amount)
                + 1.5 * (PERCENTILE_CONT(0.75) WITHIN GROUP (ORDER BY claim_amount)
                       - PERCENTILE_CONT(0.25) WITHIN GROUP (ORDER BY claim_amount)) AS upper_fence
        FROM claims WHERE claim_amount IS NOT NULL AND claim_amount > 0
    )

    FROM claims c CROSS JOIN bounds b
    WHERE c.claim_amount IS NOT NULL;

        SELECT

            PERCENTILE_CONT(0.99) WITHIN GROUP (ORDER BY claim_amount) AS upper_fence
        FROM claims WHERE claim_amount IS NOT NULL AND claim_amount > 0

    SELECT c.claim_id, ROUND(c.claim_amount,2) AS original_amount,
           ROUND(CASE
               WHEN c.claim_amount > b.upper_fence THEN b.upper_fence
               WHEN c.claim_amount < b.lower_fence THEN b.lower_fence
               ELSE c.claim_amount END, 2) AS capped_amount
    FROM claims c CROSS JOIN bounds b
    WHERE c.claim_amount IS NOT NULL;
     
     
    -- ============================================================
    -- SECTION 3: WINDOW FUNCTIONS — 8 INSURANCE SCENARIOS
    -- ============================================================
     
     
    -- ── 3A. ROW_NUMBER — Single top claim per claimant ────────────
     
    WITH ranked AS (
        SELECT claim_id, claimant_id, claim_type, claim_amount,
               ROW_NUMBER() OVER (
                   PARTITION BY claimant_id
                   ORDER BY claim_amount DESC
               ) AS rn
        FROM claims
        WHERE claim_amount IS NOT NULL AND claim_amount > 0
    )
    SELECT claim_id, claimant_id, claim_type, ROUND(claim_amount,2) AS claim_amount
    FROM ranked
    WHERE rn = 1
    ORDER BY claim_amount DESC;
    -- Expected: one row per claimant, highest value claim selected
     
     
    -- ── 3B. RANK vs DENSE_RANK — Adjuster performance ─────────────
    -- Shows tie handling difference
     
    WITH counts AS (
        SELECT adjuster_id, COUNT(*) AS claim_count,
               ROUND(AVG(claim_amount),2) AS avg_claim
        FROM claims
        WHERE claim_amount > 0
        GROUP BY adjuster_id
    )
    SELECT adjuster_id, claim_count, avg_claim,
           RANK()       OVER (ORDER BY claim_count DESC) AS rank_with_gap,
           DENSE_RANK() OVER (ORDER BY claim_count DESC) AS rank_no_gap,
           RANK()       OVER (ORDER BY avg_claim   DESC) AS rank_by_avg_claim
    FROM counts
    ORDER BY claim_count DESC;
    -- Try: change ORDER BY to avg_claim DESC and observe how ranks shift
     
     
    -- ── 3C. LAG — Detect repeat claims within 30 days ─────────────
     
    WITH ordered AS (
        SELECT claim_id, claimant_id, fnol_date, claim_amount, claim_type,
               LAG(fnol_date)    OVER (PARTITION BY claimant_id ORDER BY fnol_date) AS prev_fnol,
               LAG(claim_amount) OVER (PARTITION BY claimant_id ORDER BY fnol_date) AS prev_amount,
               LAG(claim_type)   OVER (PARTITION BY claimant_id ORDER BY fnol_date) AS prev_type
        FROM claims
        WHERE fnol_date IS NOT NULL
    )
    SELECT claim_id, claimant_id, fnol_date, prev_fnol,
           DATEDIFF(day, prev_fnol, fnol_date) AS days_since_last,
           -- SQLite: CAST(JULIANDAY(fnol_date) - JULIANDAY(prev_fnol) AS INT)
           ROUND(claim_amount,2) AS claim_amount,
           ROUND(prev_amount, 2) AS prev_amount
    FROM ordered
    WHERE prev_fnol IS NOT NULL
      AND DATEDIFF(day, prev_fnol, fnol_date) <= 30
    ORDER BY days_since_last;
    -- Expected: claimants who filed two claims within a month — fraud signal
     
     
    -- ── 3C. LEAD — Next expected claim date ───────────────────────
     
    SELECT claim_id, claimant_id, fnol_date, claim_type,
           LEAD(fnol_date,   1) OVER (PARTITION BY claimant_id ORDER BY fnol_date) AS next_claim_date,
           LEAD(claim_type,  1) OVER (PARTITION BY claimant_id ORDER BY fnol_date) AS next_claim_type,
           LEAD(claim_amount,1) OVER (PARTITION BY claimant_id ORDER BY fnol_date) AS next_claim_amount
    FROM claims
    WHERE fnol_date IS NOT NULL
    ORDER BY claimant_id, fnol_date;
     
     
    -- ── 3D. SUM — Running total of claims paid by month ───────────
     
    WITH monthly AS (
        SELECT
            -- SQL Server / Snowflake:
            FORMAT(fnol_date, 'yyyy-MM')         AS month,
            -- SQLite: STRFTIME('%Y-%m', fnol_date)
            -- BigQuery: FORMAT_DATE('%Y-%m', fnol_date)
            ROUND(SUM(claim_amount), 2)          AS monthly_total,
            COUNT(*)                             AS claim_count
        FROM claims
        WHERE claim_amount IS NOT NULL AND claim_amount > 0
          AND fnol_date    IS NOT NULL
        GROUP BY FORMAT(fnol_date, 'yyyy-MM')
    )
    SELECT month, monthly_total, claim_count,
           ROUND(SUM(monthly_total) OVER (
               ORDER BY month
               ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
           ), 2) AS running_total,
           SUM(claim_count) OVER (
               ORDER BY month
               ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
           ) AS running_count
    FROM monthly
    ORDER BY month;
     
     
    -- ── 3E. AVG — Rolling 3-month and 12-month average ────────────
     
    WITH monthly AS (
        SELECT FORMAT(fnol_date, 'yyyy-MM') AS month,
               ROUND(AVG(claim_amount), 2)  AS avg_monthly
        FROM claims
        WHERE claim_amount IS NOT NULL AND claim_amount > 0 AND fnol_date IS NOT NULL
        GROUP BY FORMAT(fnol_date, 'yyyy-MM')
    )
    SELECT month, avg_monthly,
           ROUND(AVG(avg_monthly) OVER (
               ORDER BY month ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
           ), 2) AS rolling_3mo_avg,
           ROUND(AVG(avg_monthly) OVER (
               ORDER BY month ROWS BETWEEN 11 PRECEDING AND CURRENT ROW
           ), 2) AS rolling_12mo_avg
    FROM monthly
    ORDER BY month;
    -- Tip: ROWS BETWEEN 2 PRECEDING AND CURRENT ROW = current + 2 previous = 3 months total
     
     
    -- ── 3F. NTILE — Claimant risk quartiles ───────────────────────
     
    WITH spend AS (
        SELECT claimant_id,
               ROUND(SUM(claim_amount), 2) AS total_spend,
               COUNT(*)                    AS claim_count,
               MAX(fraud_flag)             AS any_fraud
        FROM claims
        WHERE claim_amount IS NOT NULL AND claim_amount > 0
        GROUP BY claimant_id
    )
    SELECT claimant_id, total_spend, claim_count, any_fraud,
           NTILE(4) OVER (ORDER BY total_spend DESC) AS spend_quartile,
           CASE NTILE(4) OVER (ORDER BY total_spend DESC)
               WHEN 1 THEN 'High Risk'
               WHEN 2 THEN 'Medium-High'
               WHEN 3 THEN 'Medium-Low'
               WHEN 4 THEN 'Low Risk'
           END AS risk_band
    FROM spend
    ORDER BY total_spend DESC;
    -- Then add outer GROUP BY risk_band to get avg fraud rate per band
     
     
    -- ── 3G. FIRST_VALUE & LAST_VALUE — Reserve development ────────
    -- KEY GOTCHA: LAST_VALUE needs explicit frame or returns current row!
     
    SELECT claim_id, update_date, ROUND(reserve_amount,2) AS reserve_amount,
           ROUND(FIRST_VALUE(reserve_amount) OVER (
               PARTITION BY claim_id ORDER BY update_date
               ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
           ), 2) AS initial_reserve,
           ROUND(LAST_VALUE(reserve_amount) OVER (
               PARTITION BY claim_id ORDER BY update_date
               ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
               -- WITHOUT this frame, LAST_VALUE = CURRENT ROW (common mistake!)
           ), 2) AS final_reserve
    FROM reserve_history
    ORDER BY claim_id, update_date;
     
    -- Summary: reserve movement per claim
    WITH dev AS (
        SELECT DISTINCT claim_id,
               FIRST_VALUE(reserve_amount) OVER (
                   PARTITION BY claim_id ORDER BY update_date
                   ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS initial_reserve,
               LAST_VALUE(reserve_amount)  OVER (
                   PARTITION BY claim_id ORDER BY update_date
                   ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS final_reserve
        FROM reserve_history
    )
    SELECT claim_id,
           ROUND(initial_reserve, 2) AS initial_reserve,
           ROUND(final_reserve,   2) AS final_reserve,
           ROUND((final_reserve - initial_reserve) / NULLIF(initial_reserve,0) * 100, 1) AS change_pct,
           CASE
               WHEN final_reserve > initial_reserve * 1.5 THEN 'Significantly strengthened'
               WHEN final_reserve > initial_reserve       THEN 'Strengthened'
               WHEN final_reserve < initial_reserve * 0.5 THEN 'Significantly released'
               ELSE 'Released / stable'
           END AS reserve_movement
    FROM dev
    ORDER BY change_pct DESC;
     
     
    -- ── 3H. PERCENT_RANK — Claim percentile within type ───────────
     
    SELECT claim_id, claim_type, ROUND(claim_amount,2) AS claim_amount,
           ROUND(PERCENT_RANK() OVER (
               PARTITION BY claim_type ORDER BY claim_amount
           ) * 100, 1) AS percentile_in_type,
           ROUND(CUME_DIST() OVER (
               PARTITION BY claim_type ORDER BY claim_amount
           ) * 100, 1) AS cumulative_dist_pct
    FROM claims
    WHERE claim_amount IS NOT NULL AND claim_amount > 0
    ORDER BY claim_type, percentile_in_type DESC;
     
    -- Practical: flag top 5% per type for SIU review
    SELECT *
    FROM (
        SELECT claim_id, claim_type, ROUND(claim_amount,2) AS claim_amount,
               ROUND(PERCENT_RANK() OVER (
                   PARTITION BY claim_type ORDER BY claim_amount
               ) * 100, 1) AS percentile_in_type
        FROM claims
        WHERE claim_amount IS NOT NULL AND claim_amount > 0
    ) ranked
    WHERE percentile_in_type >= 95
    ORDER BY claim_type, claim_amount DESC;
     
     
    -- ── 3I. COMBINED — Full fraud scoring query ────────────────────
     
    SELECT c.claim_id, c.claimant_id, c.claim_type,
           ROUND(c.claim_amount, 2) AS claim_amount,
     
           -- Percentile rank within claim type (high = unusual)
           ROUND(PERCENT_RANK() OVER (
               PARTITION BY c.claim_type ORDER BY c.claim_amount
           ) * 100, 1) AS pct_rank_in_type,
     
           -- Days since this claimant's previous claim (LAG)
           DATEDIFF(day,
               LAG(c.fnol_date) OVER (PARTITION BY c.claimant_id ORDER BY c.fnol_date),
               c.fnol_date
           ) AS days_since_last_claim,
     
           -- How many claims has this claimant made (running count)
           COUNT(*) OVER (
               PARTITION BY c.claimant_id
               ORDER BY c.fnol_date
               ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
           ) AS claim_sequence,
     
           -- Composite risk score
           CASE WHEN PERCENT_RANK() OVER (
                   PARTITION BY c.claim_type ORDER BY c.claim_amount) >= 0.95
                THEN 3 ELSE 0 END
           + CASE WHEN DATEDIFF(day,
                   LAG(c.fnol_date) OVER (PARTITION BY c.claimant_id ORDER BY c.fnol_date),
                   c.fnol_date) <= 30
                THEN 3 ELSE 0 END
           + CASE WHEN c.fraud_flag = 1 THEN 2 ELSE 0 END
           + CASE WHEN c.num_prev_claims > 4 THEN 1 ELSE 0 END AS fraud_risk_score
     
    FROM claims c
    WHERE c.claim_amount IS NOT NULL AND c.claim_amount > 0
    ORDER BY fraud_risk_score DESC, claim_amount DESC;
     
     
    -- ============================================================
    -- SECTION 4: INSURANCE METRICS
    -- ============================================================
     
     
    -- ── 4A. LOSS RATIO ─────────────────────────────────────────────
     
    SELECT claim_type, COUNT(*) AS claim_count,
           ROUND(SUM(paid_amount), 2)  AS total_paid,
           ROUND(SUM(premium),     2)  AS total_premium,
           ROUND(100.0 * SUM(paid_amount) / NULLIF(SUM(premium),0), 1) AS loss_ratio_pct,
           CASE
               WHEN 100.0 * SUM(paid_amount) / NULLIF(SUM(premium),0) > 100 THEN 'Unprofitable'
               WHEN 100.0 * SUM(paid_amount) / NULLIF(SUM(premium),0) >  70 THEN 'Watch'
               ELSE 'Healthy'
           END AS loss_ratio_status
    FROM claims
    WHERE paid_amount IS NOT NULL AND premium IS NOT NULL
    GROUP BY claim_type
    ORDER BY loss_ratio_pct DESC;
     
     
    -- ── 4B. CYCLE TIME ─────────────────────────────────────────────
     
    SELECT claim_type, COUNT(*) AS closed_claims,
           ROUND(AVG(DATEDIFF(day, fnol_date, closure_date)), 1) AS avg_cycle_days,
           ROUND(PERCENTILE_CONT(0.5) WITHIN GROUP
                 (ORDER BY DATEDIFF(day, fnol_date, closure_date)), 1) AS median_cycle_days,
           MIN(DATEDIFF(day, fnol_date, closure_date)) AS min_days,
           MAX(DATEDIFF(day, fnol_date, closure_date)) AS max_days
    FROM claims
    WHERE claim_status = 'Closed' AND closure_date IS NOT NULL AND fnol_date IS NOT NULL
    GROUP BY claim_type
    ORDER BY avg_cycle_days DESC;
     
     
    -- ── 4C. COHORT — Monthly resolution rate ───────────────────────
     
    WITH cohort AS (
        SELECT FORMAT(fnol_date, 'yyyy-MM') AS cohort_month,
               COUNT(*) AS total_claims,
               SUM(CASE WHEN claim_status = 'Closed' THEN 1 ELSE 0 END) AS closed_claims
        FROM claims WHERE fnol_date IS NOT NULL
        GROUP BY FORMAT(fnol_date, 'yyyy-MM')
    )
    SELECT cohort_month, total_claims, closed_claims,
           ROUND(100.0 * closed_claims / NULLIF(total_claims,0), 1) AS resolution_rate_pct,
           SUM(total_claims) OVER (
               ORDER BY cohort_month
               ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
           ) AS running_total
    FROM cohort
    ORDER BY cohort_month;
     
     
    -- ── 4D. PROVIDER FRAUD ANALYSIS ────────────────────────────────
     
    SELECT provider_id, COUNT(*) AS total_claims,
           SUM(fraud_flag) AS fraud_count,
           ROUND(100.0 * SUM(fraud_flag) / NULLIF(COUNT(*),0), 1) AS fraud_rate_pct,
           ROUND(AVG(claim_amount),2) AS avg_claim_amount,
           ROUND(SUM(claim_amount),2) AS total_claimed,
           DENSE_RANK() OVER (ORDER BY SUM(claim_amount) DESC) AS spend_rank
    FROM claims
    WHERE provider_id IS NOT NULL AND claim_amount > 0
    GROUP BY provider_id
    ORDER BY total_claimed DESC;
