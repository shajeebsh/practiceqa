**Du’aa: “My Lord! I seek refuge with You from the whisperings of the Shayaateen”**

[![quranic-dua-23-9798](https://muslimfortress.files.wordpress.com/2011/11/quranic-dua-23-97981.jpg?w=302&h=102)](<https://muslimfortress.files.wordpress.com/2011/11/quranic-dua-23-97981.jpg>)

> **“My Lord! I seek refuge with You from the whisperings of the Shayaateen.  And I seek refuge with You, My Lord! lest they should come near me.”  
> ** _(Surah Al Muminoon 23:97-98)_

![](https://shajjjji.wordpress.com/wp-content/uploads/2020/07/whatsapp-image-2020-07-27-at-8.26.39-am.jpeg?w=476) ![](https://shajjjji.wordpress.com/wp-content/uploads/2020/07/whatsapp-image-2020-07-15-at-9.15.12-am.jpeg?w=627) **بِاسْمِكَ رَبِّي وَضَعْتُ جَنْبِي، وَبِكَ أَرْفَعُهُ، فَإِنْ أَمْسَكْتَ نَفْسِي فَارْحَمْهَا، وَإِنْ أَرْسَلْتَهَا فَاحْفَظْهَا، بِمَا تَحْفَظُ بِهِ عِبَادَكَ الصَّالِحِينَ**  
---  
With Your Name(1) my Lord, I lay myself down ; and with Your Name I rise . And if my soul You take , have mercy on it , and if You send it back then protect it as You protect Your righteous slaves.(2)

 **🌅🕋 *اَللَّه‍ُمَّ صَلِّ عَلَى سَيِّدِنَا مُحَمَّدٍ عَبْدِكَ وَنَبِيِّكَ وَرَسُولِكَ النَّبِيِّ اْلأُمِّيِّ وَ عَلَى آلِهِ وَاَزْوَاجِهِ وَذُرِّيَّتِهِ وَسَلِّمْ عَدَدَ خَلْقِكَ وَرِضَا نَفْسِكَ وَزِنَةَ عَرْشِكَ وَمِدَادَ كَلِمَاتِك*ْ**

***സ്വലാത്തുന്നാരിയ്യഃ***  
 **🌅🕋 *اللَّهُمَّ صَلِّ صَلاَةً كَامِلَةً وَسَلِّمْ سَلاَماً تَامًّا عَلَى سَيِّدِنَا مُحَمَّد ٍالذي تَنْحَلُّ بِهِ الْعُقَدُ وَتَـنْفَرِجُ بِهِ الْكُرَبُ وَتُـقْضَى بِهِ الْحَوَائِجُ وَتُنَالُ بِهِ الرَّغَائِبُ وَحُسْنُ الْخَوَاتِمِ وَيُسْتَسْقَى الْغَمَامُ بوَجْهِهِ الْكَرِيمِ وَعَلى آلِهِ وَصَحْبِهِ فِي كُـلِّ لَمْحَةٍ وَنَفَسٍ بِعَدَدِ كُـلِّ مَعْلُومٍ لَك***

***ഇസ്മുൽ അഅ്ളം സ്വലാത്ത്***  
 **🌅🕋 *اَللَّهُمَّ اِنِّي اَسْئَلُكَ بِسْمِكَ اْلأَعْظَمِ الْمَكْتُوبِ مِنْ نُورِ وَجْهِكَ الْأَعْلَى الْمُأَبَّدِ الدَّائِمِ الْبَاقِي الْمُخَلَّدِ فِي قَلْبِ نَبِيِّكَ وَرَسُولِكَ سَيِّدِنَا مُحَمَّدٍ(ص)وَاَسْئَلُكَ بِسْمِكَ اْلأَعْظَمِ الْوَاحِدِ بِوَحْدَةِ الْأَحَدِ الْمُتَعَالِي عَنْ وَحْدَةِالْكَمِّ وَالْعَدَدِ الْمُقَدَّسِ عَنْ كُلِّ اَحَدْ.***

***وَبِحَقِّ بِسْمِ اللَّهِ الرَّحْمٰنِ الرَّحِيمْ قُلْ ه‍ُوَ اللَّهُ اَحَدْ اَللَّهُ الصَّمَدْ لَمْ يَلِدْ وَلَمْ يُولَدْ وَلَمْ يَكُنْ لَهُ كُفُوًا اَحَدْ.***

***اَنْ تُصَلِّيَ عَلَى سَيِّدِنَا مُحَمَّدٍ سِرِّ حَيٰوةِ الْوُجُودِ وَسَبَبِ الْأَعْظَمِ لِكُلِّ مَوْجُودٍ صَلاَةً تُثَبِّتُ فِي قَلْبِي الْإِيمَانِ وَتُحَفِّظُنِي الْقُرْآنَ وَتُفَهِّمُنِي مِنْهُ الْآيَاتِ وَتَفْتَحُ لِي بِهَا نُورَالْجَنَّاتِ وَنُورَالنَّعِيمِ وَنُورَ النَّظَرِاِلَى وَجْهِكَ الْكَرِيمْ وَعَلَى آلِهِ وَ صَحْبِهِ وَ سَلِّمْ.***

 ***ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْعَابِدِين* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْحَامِدِين***  
 ***ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالصّٰالِحِينْ.* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالرٰاكِعِين***  
 **. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالسّٰاجِدِينْ*. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْقٰۤاءِمِين*. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالقٰاعِدِين***  
 ***ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْمُتَّقِين*ْ. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْمُسْتَغْفِرِين***  
 ***ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالنّٰادِمِين* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالشّٰاكِرِين*.**  
***ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّد ِالْخٰافِضِين* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالذّٰكِرِينْ.***  
 ***ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْعٰاقِلِينْ*. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِ الْمُحْسِنِينْ*. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِ الْاَ كْرَمِين* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْمُنْذِرِينْ.* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْمُبَشِّرِينْ.* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالطَّيِّبِينْ*. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالنَّبِيِّين* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ سَيِّدِالْعٰالَمِينْ*. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ نِ النَّبِيِّ الزَّكِيِّا لتَّقِي*. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ نِ الْقُرَشِيِّ الْهٰاشِمِيِّ.* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ نِ الْمَدِنِيِّ الْعَرَبِيِّ الْمُكَرَّمِ يَوْمَ الْقِيٰمَةِ.* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪ ٍسَيِّدِ اَهْلِ الْجَنَّةِ.* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍصٰاحِبَ الْمَقٰامِ الْمَحْمُود* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍصٰاحِبَ الصِّرَاطِ الْمُسْتَقِينْ.* *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍاَفَضَلِ اْلاَوَلِينَ وَاْلاَخِرِين*ْ. *ﺍَﻟﻠَّﻬُﻢَّ ﺻَﻞِّ وَسَلِّمْ ﻋَﻠَﻰ ﺳَﻴِّﺪِﻧَﺎ ﻣُﺤَﻤَّﺪٍ وَعَلٰي جَمِيعِ الْمَلٰئِكَةِ الْمُقَرَّبِينْ. وَعَلٰي عِبَادِاللّٰهِ الصّٰالِحِينْ*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ ذٰاكِرًاوَحَبِيبًا وَمُذَكِّرً امُحَمَّدٍ رَسُولِ الّٰله*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ اَحْمَدً اوَمُحَمَّدً اوَسَيِّدً مُحَمَّدٍ رَّسُولالّٰلهِ *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ صٰابِرً اوَنَبِيًّا وَمُرَاقِبًا مُحَمَّدٍ رَسُيلِ الّٰلهِ*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ عٰالِيًا وَرَحِيمًا مُحَمَّدٍ رَسُولِ الّٰلهِ*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ عٰاقِبًاوَ كَرِيمًا مُحَمَّدٍ رَسُولِ الّٰله* *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ عَدْلاً وَجَوٰادً وَمُزَمِّلاً مُحَمَّدٍ رَسُولِ الّٰله.ِ* *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ قٰاسِمًا وَمَهْدِيًّا وَهٰادِيًا مُحَمَّدٍ رَسُولِ الّٰله.* *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ شَكُورًا وَحَرِيصًا مُحَمَّدٍ رَسُولِ الّٰله.* *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ قٰۤا ئِمًا وَصَفِيًّا وَعَبْدَ الّٰلهِ مُحَمَّدٍ رَسُولِ الّٰله*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ شٰاهِدً ابَصِيرً ا وَمَهْدِيًّا مُحَمَّدٍ رَسُولِ الّٰله*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ بٰاهِيًا وَنُورً ا مَكِّيًّا مُحَمَّدٍ رَسُولِ الّٰله*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ شٰاكِرً ا وَوَ لِيًّا وَ نَذِيرً ا مُحَمَّدٍ رَسُولِ الّٰله.* *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ بُرْهٰانًا صَحِيحًا وَ شَرِيفًا مُحَمَّدٍ رَسُولِ الّٰله*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ طَاهِرًا وَصَفِيًّا و َمُخْتٰارًا مُحَمَّدٍ رَسُولِ الّٰله*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ مُسْلِمًا وَرَؤُفًا مُحَمَّدٍ رَسُولِ الّٰله. اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ مُؤْمِنًا وَ حَلِيمًا مُحَمَّدٍ رَسُولِ الّٰله*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ قٰائِمًا وَمَحْمُودًا وَحَامِدًا مُحَمَّدٍ رَسُولِ الّٰله*. *اَلَّلهُمَّ صَلِّ وَسَلِّمْ عَلٰي مَنْ سَمَّيْتَهُ مِصْبَاحًا وَآمِرًا وَنٰاهِيًا مُحَمَّدٍ رَسُولِ الّٰلهِ صَلَّي الّٗلهُ عَلَيْهِ وَعَلٰي آلِهِ وَصَحْبِهِ وَاَزْوٰاجِهِ وَذُرِّيّٰاتِهِ وَاَهْلِ بَيْتِهِ وَعَنْ كُلِّ اصَّحٰابَةِ اَجْمَعِينَ اٰمِين*ْ.**

***لَوْ أَنزَلْنَا هَٰذَا الْقُرْآنَ عَلَىٰ جَبَلٍ لَّرَأَيْتَهُ خَاشِعًا مُّتَصَدِّعًا مِّنْ خَشْيَةِ اللَّهِ ۚ وَتِلْكَ الْأَمْثَالُ نَضْرِبُهَا لِلنَّاسِ لَعَلَّهُمْ يَتَفَكَّرُونَ ۞ هُوَ اللَّهُ الَّذِي لَا إِلَٰهَ إِلَّا هُوَ ۖ عَالِمُ الْغَيْبِ وَالشَّهَادَةِ ۖ هُوَ الرَّحْمَٰنُ الرَّحِيمُ ۞ هُوَ اللَّهُ الَّذِي لَا إِلَٰهَ إِلَّا هُوَ الْمَلِكُ الْقُدُّوسُ السَّلَامُ الْمُؤْمِنُ الْمُهَيْمِنُ الْعَزِيزُ الْجَبَّارُ الْمُتَكَبِّرُ ۚ سُبْحَانَ اللَّهِ عَمَّا يُشْرِكُونَ ۞ هُوَ اللَّهُ الْخَالِقُ الْبَارِئُ الْمُصَوِّرُ ۖ لَهُ الْأَسْمَاءُ الْحُسْنَىٰ ۚ يُسَبِّحُ لَهُ مَا فِي السَّمَاوَاتِ وَالْأَرْضِ ۖ وَهُوَ الْعَزِيزُ الْحَكِيمُ ۞***  
 ***(الحشر : 21-24)***

*രാത്രി ഓതിയാൽ രാവിലെ വരെ അവന്ന് വേണ്ടി 70,000 മലക്കുകൾ പൊറുക്കലിനെ തേടും, മാത്രവുമല്ല അവന്ന് ശഹീദിന്റെ പ്രതിഫലവും ലഭിക്കും

നാളെ സഫർ അവസാന ബുധൻ ആപത്തുകൾ ഭൂമിയിലേക് ഇറങ്ങുന്ന ദിവസം ആണ്, അന്ന് രാത്രിയും രാവിലെയും ചൊല്ലേണ്ട ദിഖ്‌ർ
