## 🗄️ 4. BigQuery SQL Patterns

## Arrays & UNNEST
    
    
    sql _-- Generate numbers 10-20_
    SELECT num FROM UNNEST(GENERATE_ARRAY(10, 20)) AS num
    
    _-- Demo table with dollars_
    WITH base_table AS (
      SELECT dollars, index
      FROM UNNEST(GENERATE_ARRAY(1,5)) dollars 
      WITH OFFSET AS index
    )
    

## Running Totals & Moving Averages
    
    
    sql _-- Running total (correlated subquery)_
    SELECT bt.*, 
      (SELECT SUM(bt2.dollars) 
       FROM base_table bt2 
       WHERE bt2.index <= bt.index) AS running_total
    FROM base_table bt
    
    _-- 3-row moving average_
    SELECT *, AVG(dollars) OVER (
      ORDER BY index 
      ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
    ) AS ma_3
    FROM base_table
    

## First Order Analysis
    
    
    sqlWITH first_orders AS (
      SELECT customer_id, amount, payment_date,
        ROW_NUMBER() OVER (PARTITION BY customer_id ORDER BY payment_date) rn
      FROM payments
      WHERE rn = 1
    )
    SELECT p.customer_id, f.amount first_order,
           SUM(p.amount) total_rev,
           f.amount/SUM(p.amount) first_pct_total
    FROM payments p JOIN first_orders f ON p.customer_id = f.customer_id
    GROUP BY 1,2

## Median (No Native Function)
    
    
    sqlWITH ranked AS (
      SELECT OrderTotal, ROW_NUMBER() OVER (ORDER BY OrderTotal) rn,
             COUNT(*) OVER() cnt
      FROM Orders
    )
    SELECT AVG(OrderTotal)
    FROM ranked 
    WHERE rn IN (FLOOR((cnt+1)/2.0), CEILING((cnt+1)/2.0))
